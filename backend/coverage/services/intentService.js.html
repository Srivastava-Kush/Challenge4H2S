function resolveNodeId(raw) {
  if (!raw) return null;
  // Already a valid node ID (contains underscore)
  if (raw.includes('_')) return raw;
  const lower = raw.toLowerCase().trim();
  return NODE_ALIASES[lower] || null;
}
